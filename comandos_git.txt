$ git init --> inicio de repositorio como una bd
$ git add nombredeltexto.txt -->etiquetar el archivo para que el 
			        repositorio sepa que esta

$ git commit -m "comentarios" --> envia los cambios a la bd del repositorio
			          para que se guarde de forma permanente

$git branch nombredelarama --> crea una rama

$git branch --list --> muestra las ramas creadas

$git checkout nombredelarama --> ingresamos a la rama creada

$git merge nombredelaramadondetrabajo --> Une las dos ramas por ejemplo estando en master 		
					ingreso el comando marge uno la rama en la que trabajo	
					al master

$git remote add origin larutadelrepositorioengithub --> este comando sirve para lograr la comunicacion	
							del git local con el github remoto
